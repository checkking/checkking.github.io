Golang初始化顺序

Nov 5, 2018

一个例子

package main

import (
    "fmt"
)

func init() {
    a := "xxx"
    fmt.Printf("hello, %s, %s", "world!", a)
}

func main() {
    fmt.Println("main...")
}
